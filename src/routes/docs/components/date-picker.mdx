## Installation

### CLI

```package-exec
shadcn@latest add @gc-solid/date-picker
```

### Manual

Install the following dependencies:

```package-install
@kobalte/core @corvu/calendar
```

Copy and paste the following code into your project.

```file="~/registry/ui/date-picker.tsx" showLineNumbers

```

Also install the [Calendar](/docs/components/calendar), [Popover](/docs/components/popover), and [Button](/docs/components/button) components.

## Usage

```tsx
import {
  DatePicker,
  DatePickerCalendar,
  DatePickerContent,
  DatePickerInput,
  DatePickerTrigger,
  formatDate,
} from "~/components/ui/date-picker";
```

```tsx
<DatePicker>
  <DatePickerTrigger>
    <DatePickerInput placeholder="Pick a date">
      <Show when={date()}>
        {(d) => <span>{formatDate(d())}</span>}
      </Show>
    </DatePickerInput>
  </DatePickerTrigger>
  <DatePickerContent>
    <DatePickerCalendar
      mode="single"
      value={date()}
      onValueChange={(value) => setDate(value)}
    />
  </DatePickerContent>
</DatePicker>
```
