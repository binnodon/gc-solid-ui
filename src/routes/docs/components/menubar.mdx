## Installation

### CLI

```package-exec
shadcn@latest add @gc-solid/menubar
```

### Manual

Install the following dependencies:

```package-install
@kobalte/core lucide-solid
```

Copy and paste the following code into your project.

```file="~/registry/ui/menubar.tsx" showLineNumbers

```

## Usage

```tsx
import {
  Menubar,
  MenubarContent,
  MenubarItem,
  MenubarMenu,
  MenubarTrigger,
} from "~/components/ui/menubar";
```

```tsx
<Menubar>
  <MenubarMenu>
    <MenubarTrigger>File</MenubarTrigger>
    <MenubarContent>
      <MenubarItem>New</MenubarItem>
      <MenubarItem>Open</MenubarItem>
    </MenubarContent>
  </MenubarMenu>
</Menubar>
```
